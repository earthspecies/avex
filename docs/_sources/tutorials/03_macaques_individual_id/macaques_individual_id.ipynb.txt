    "MIN_SAMPLES = int(0.5 * TARGET_SR)\n",
    "\n",
    "\n",
    "def load_audio(path: str, target_sr: int = TARGET_SR) -> torch.Tensor:\n",
    "    \"\"\"Load a WAV, convert to mono, resample, zero-pad to at least 0.5 s.\"\"\"\n",
    "    wav, sr = sf.read(path, dtype=\"float32\", always_2d=True)\n",
    "    wav = wav.mean(axis=1)\n",
    "    if sr != target_sr:\n",
    "        wav = librosa.resample(wav, orig_sr=sr, target_sr=target_sr)\n",
    "    if len(wav) < MIN_SAMPLES:\n",
    "        wav = np.pad(wav, (0, MIN_SAMPLES - len(wav)))\n",
    "    return torch.from_numpy(wav).unsqueeze(0)\n",
    "\n",
    "\n",
    "print(f\"Sample shape: {load_audio(df['path'].iloc[0]).shape}\")"

    "BEATS_ALL_CACHE = EMBED_DIR / \"beats_all_layers_embeddings.npy\"\n",
    "\n",
    "if BEATS_ALL_CACHE.exists() and np.load(BEATS_ALL_CACHE).shape[0] == len(df):\n",
    "    beats_all_embs = np.load(BEATS_ALL_CACHE)\n",
    "    print(f\"Loaded cached BEATs all-layers embeddings: {beats_all_embs.shape}\")\n",
    "else:\n",
    "    print(f\"Loading model for all-layers extraction: {BEATS_MODEL}\")\n",
    "    model = load_model(BEATS_MODEL, return_features_only=True, device=DEVICE)\n",
    "    model.eval()\n",
    "\n",
    "    # Register forward hooks on every transformer encoder layer\n",
    "    try:\n",
    "        encoder_layers = model.model.encoder.layers\n",
    "    except AttributeError:\n",
    "        encoder_layers = model.backbone.encoder.layers\n",
    "    n_layers = len(encoder_layers)\n",
    "    layer_store: dict = {}\n",
    "    hooks = []\n",
    "    for i, layer in enumerate(encoder_layers):\n",
    "\n",
    "        def _make_hook(idx):\n",
    "            def _hook(module, inp, out):\n",
    "                layer_store[idx] = out[0] if isinstance(out, tuple) else out\n",
    "\n",
    "            return _hook\n",
    "\n",
    "        hooks.append(layer.register_forward_hook(_make_hook(i)))\n",
    "    print(f\"  Registered hooks on {n_layers} transformer layers.\")\n",
    "\n",
    "    all_embs = []\n",
    "    with torch.no_grad():\n",
    "        for path in tqdm(df[\"path\"], desc=\"BEATs all-layers\"):\n",
    "            layer_store.clear()\n",
    "            wav = load_audio(path)\n",
    "            _ = model(wav)\n",
    "            # Mean-pool each layer then average across layers → (D,)\n",
    "            # Handles both time-first (T, B, D) and batch-first (B, T, D) outputs\n",
    "            per_layer = [\n",
    "                layer_store[i].view(-1, layer_store[i].shape[-1]).mean(dim=0).cpu().numpy() for i in range(n_layers)\n",
    "            ]\n",
    "            all_embs.append(np.mean(per_layer, axis=0))\n",
    "\n",
    "    for h in hooks:\n",
    "        h.remove()\n",
    "\n",
    "    beats_all_embs = np.stack(all_embs)\n",
    "    np.save(BEATS_ALL_CACHE, beats_all_embs)\n",
    "    print(f\"Saved BEATs all-layers embeddings: {beats_all_embs.shape}\")\n",
    "    del model"
